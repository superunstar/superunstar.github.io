# !! Ne PAS utiliser ce script pour générer des clés RSA. Ce script est uniquement à but éducatif, afin de comprendre le fonctionnement de la génération de clés RSA, il n'est en aucun cas suffisament sécurisé pour pouvoir être utilisé en condition réelles.

import random
from math import gcd

def is_prime(num):
    if num < 2:
        return False
    for i in range(2, int(num**0.5) + 1):
        if num % i == 0:
            return False
    return True

def generate_prime_candidate(length):
    p = random.getrandbits(length)
    # Appliquer un masque pour définir le MSB et le LSB sur 1
    p |= (1 << length - 1) | 1
    return p

def generate_prime_number(length=1024):
    p = 4
    while not is_prime(p):
        p = generate_prime_candidate(length)
    return p

def mod_inverse(e, phi):
    def egcd(a, b):
        if a == 0:
            return (b, 0, 1)
        g, y, x = egcd(b % a, a)
        return (g, x - (b // a) * y, y)
    g, x, y = egcd(e, phi)
    if g != 1:
        raise Exception('Inverse modulaire non existant')
    return x % phi

def generate_keypair(bits):
    p = generate_prime_number(bits)
    q = generate_prime_number(bits)
    n = p * q
    phi = (p - 1) * (q - 1)

    e = random.randrange(1, phi)
    g = gcd(e, phi)
    while g != 1:
        e = random.randrange(1, phi)
        g = gcd(e, phi)

    d = mod_inverse(e, phi)

    return ((e, n), (d, n))

def main():
    bits = 1024  # Taille des clés
    public, private = generate_keypair(bits)
    print(f"Clé publique : {public}")
    print(f"Clé privée : {private}")

if __name__ == '__main__':
    main()
